parseExpression → parseOrExpr

parseOrExpr → parseAndExpr ( "or" parseAndExpr )*
parseAndExpr → parseNotExpr ( "and" parseNotExpr )*
parseNotExpr → "not" parseNotExpr | parseRelExpr
parseRelExpr → parseArithExpr ( REL_OP parseArithExpr )*
parseArithExpr → parseTerm ( (+|-) parseTerm )*
parseTerm → parsePower ( (*|/|%) parsePower )*
parsePower → parseUnary (** parsePower)?
parseUnary → (+|-|~) parseUnary | parseFactor
parseFactor → (expr) | ID | NUM
